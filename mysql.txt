CREATE DATABASE servicos;
USE servicos;

-- Criar tabela
CREATE TABLE IF NOT EXISTS eventos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    data_evento DATE NOT NULL,
    local VARCHAR(100) NOT NULL,
    descricao VARCHAR(255) NOT NULL
);
